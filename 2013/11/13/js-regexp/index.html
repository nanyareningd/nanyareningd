javascript：正则表达式

这是一段检测浏览器名称及版本号的代码，里面涉及到正则表达式的匹配

function detectBrowser(){
    var ua = navigator.userAgent.toLowerCase();
    var reg = /(opera|firefox|chrome|safari|msie|trident(?=\/))\/?\s*([\d.]+)/i;
    var result = reg.exec(ua);
    if(result[1] == "trident"){result[2] = ua.match(/\brv\:\s*([\d.]+)/)[1];result[1]="IE";}
    if(result[1] == "safari"){result[2] = ua.match(/\bversion\/([\d.]+)/i)[1];}
    return [result[1],result[2]];
}
alert('浏览器:'+detectBrowser()[0]+" "+detectBrowser()[1]);
